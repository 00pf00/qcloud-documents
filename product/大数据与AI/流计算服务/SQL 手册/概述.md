```
CREATE TABLE `MySQLSourceTable` (
    `id` bigint,
    `name` varchar(30),
     age int,
    `weight` double,
    newColumnStr1 string,
    PRIMARY KEY (`id`) NOT ENFORCED -- 如果要同步的数据库表定义了主键, 则这里也需要定义
) WITH (
    'connector' = 'mysql-cdc',		
    'hostname' = '',   -- 数据库的 IP
    'port' = '',                -- 数据库的访问端口
    'username' = '',        -- 数据库访问的用户名（需要提供 SHOW DATABASES, REPLICATION SLAVE, REPLICATION CLIENT, SELECT, RELOAD 权限）
    'password' = '',    -- 数据库访问的密码
    'database-name' = '',   -- 需要同步的数据库
    'table-name' = '',      -- 需要同步的数据表名
    'debezium.min.row.count.to.stream.results' = '10'
	)
```
